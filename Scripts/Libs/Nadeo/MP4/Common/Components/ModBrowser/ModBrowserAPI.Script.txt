Void FetchMods(Text _Enviro)
{
	if (G_TaskResult != Null) DataFileMgr.TaskResult_Release(G_TaskResult.Id);
	G_TaskResult = DataFileMgr.Media_GetGameList(C_MediaType, _Enviro ^ C_BasePath2, True, False, False);
}

Boolean HasFetchedMods()
{
	return G_TaskResult.HasSucceeded;
}

Boolean HasAMod()
{
	return G_TaskResult.Files.count > 0;
}
